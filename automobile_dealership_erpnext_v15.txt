5.2 Vehicle Controller — Key Business Logic
class Vehicle(Document):
    def before_save(self):
        self.calculate_days_in_stock()
        self.validate_vin()
    def calculate_days_in_stock(self):
        if self.oem_invoice_date:
            self.days_in_stock = date_diff(today(), self.oem_invoice_date)
    def validate_vin(self):
        if self.vin_number and len(self.vin_number) != 17:
            frappe.throw("VIN Number must be exactly 17 characters.")
    def on_submit(self):
        if self.status == "Sold":
            self.update_inventory_ledger()

6.2 Marketplace Sync — CarDekho Integration
Vehicle listings are auto-synced to CarDekho's partner portal when a vehicle is marked Available. The
sync includes VIN, specs, price, and images. Failed syncs are logged to the Frappe error log for retry.
def sync_to_cardekho(vehicle_name):
    vehicle = frappe.get_doc("Vehicle", vehicle_name)
    settings = frappe.get_single("Auto Dealer Settings")
    payload = {
        "dealer_id": settings.cardekho_dealer_id,
        "vin": vehicle.vin_number,
        "make": vehicle.make,
        "model": vehicle.model,
        "price": vehicle.ex_showroom_price,
        # ... more fields
    }
    response = requests.post(
        settings.cardekho_api_url,
        json=payload,
        headers={"Authorization": f"Bearer {settings.cardekho_api_key}"},
    )
    response.raise_for_status()

8. Module 4 — Finance, Insurance & Billing
8.1 Loan Application — EMI Calculation
The standard reducing-balance EMI formula is used: EMI = P × r × (1+r)^n / ((1+r)^n - 1). All calculation
fields (EMI, total payable, total interest) are auto-populated on save.
def calculate_emi(self):
    p = self.loan_amount        # Principal
    r = (self.interest_rate / 100) / 12   # Monthly rate
    n = self.tenure_months
    if r > 0:
        emi = p * r * ((1 + r) ** n) / (((1 + r) ** n) - 1)
        self.emi_amount   = round(emi, 2)
        self.total_payable = round(emi * n, 2)
        self.total_interest = round(self.total_payable - p, 2)
